 39 *Cleaning the table, define Char "NA" as missing value which could be interpreted by SAS;
 40 
 41 %macro DefineNAs(want,have);
 42 DATA  &want; 
 43 SET &have;
 44 array CHAR _character_ ; 
 45     array NUM _numeric_ ;
 46 do over CHAR; 
 47     if CHAR="NA" then call missing(CHAR);
 48     else if missing(CHAR) then CHAR="WAS MISSING";
 49 end;
 50 do over NUM; 
 51     if NUM=1 then call missing(NUM);
 52     else if missing(NUM) then NUM=0;
 53 end;
 54 run;
 55 %mend ;

 61 *Using PROC SQL to JOIN Student records and Student Course table and select entries 
 62 that only related to STATS250(Intro to Statistics);
 63 PROC SQL;
 64   *TITLE "AggregatedStudentDataforSTAT250";
 65   CREATE TABLE AIM.STATS250 AS
 66   SELECT *
 67     FROM Aim.SR INNER JOIN Aim.SC
 68       ON SC.ANONID = SR.ANONID
 69     WHERE SC.SUBJECT = 'STATS' AND SC.CATALOG_NBR = 250
 70     ORDER BY SR.ANONID ASC;
 71 QUIT;

109 /*     Create a macro to do EDA, regression, and residual plotting      */
110 /*Retrieved from http://www.stat.cmu.edu/~hseltman/SASworkshop/macro.sas*/
111 
112 %macro regAndPlot(data, y, x);
113     /* Double quotes are needed to allow substitution */
114     TITLE "EDA of &data: &y on &x";
115     /* RL is regression, linear for adding a fit line to a plot */
116 
117   /*Creat scatter plot*/ 
118     SYMBOL INTERPOL=RL VALUE=plus;
119     PROC GPLOT DATA=&data;
120       PLOT &y * &x;
121     RUN;
122   /*Creat univerate leastsquare regression solution*/
123     TITLE "Regression analysis for &data";
124     PROC GLM DATA=&data;
125       MODEL &y = &x / SOLUTION;
126       OUTPUT OUT=_temp RESIDUAL=res PREDICTED=pred;
127     RUN;
128   /*Calculate residual for later use*/
129     TITLE;
130     PROC UNIVARIATE DATA=_temp NOPRINT;
131       VAR res;
132       OUTPUT OUT=_tmpstd STD=stdev;
133     RUN;
134 
135     DATA _NULL_;
136       SET _tmpstd;
137       CALL SYMPUT('resSD',stdev);
138     RUN;
139 
140   /*Q-Q plot for residual*/
141     TITLE "Regression with &data data: &y on &x";
142     TITLE2 'Quantile normal plot of residuals';
143     PROC UNIVARIATE DATA=_temp NOPRINT;
144       VAR res;
145       QQPLOT / NORMAL (MU=0 SIGMA=&resSD COLOR=red);
146     RUN;
147   /**/
148     TITLE2 'Residual vs. fit plot';
149     PROC GPLOT DATA=_temp;
150       PLOT res*pred / VREF=0;
151     RUN;  
152 %mend regAndPlot;
